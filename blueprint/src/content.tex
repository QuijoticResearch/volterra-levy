\chapter{Volterra Kernel and Combined Energy Space}

This layer defines the causal kernel $K$, the combined energy space $\mathcal{H}_{VL} = H_W \times \tilde{H}_J$, the combined divergence $\delta_{VL}$, and proves the splitting $D_{VL} = (D_W, \tilde{D}_J)$. The splitting is the key structural fact: $D$ inherits its decomposition from the energy space, not from any Gaussian assumption~[1].

\section{The Volterra Kernel}

\begin{defn}[Volterra kernel]\label{def:VolterraKernel}
\lean{VolterraKernel}
\leanok
A causal kernel $K : [0,T]^2 \to \mathbb{R}$ with $K(t,s) = 0$ for $s > t$, terminal time $T > 0$, and $L^2$-integrability: $\int_0^t K(t,s)^2\,ds < \infty$ for all $t \in [0,T]$. The Volterra--L\'evy process is $V_t = \int_0^t K(t,s)\,dL_s$, where $L$ is a L\'evy driver with both continuous and jump components~[1, Definition~3.1].
\end{defn}

\begin{defn}[Fractional kernel]\label{def:fractionalKernel}
\lean{CombinedEnergySpace.fractionalKernel}
\leanok
$K_H(t,s) = (t-s)^{H-1/2}$ for $s < t$, zero otherwise. The fractional kernel with Hurst parameter $H$ governs rough volatility: $H < 1/2$ gives rougher-than-Brownian paths, $H = 1/2$ recovers the Markov (classical) case~[1, Example~2.2].
\end{defn}

\begin{theorem}[Fractional kernel is causal]\label{thm:fractionalKernel_causal}
\lean{CombinedEnergySpace.fractionalKernel_causal}
\leanok
\uses{def:fractionalKernel}
$K_H(t,s) = 0$ for $s > t$. Causality: the kernel looks backward only.
\end{theorem}
\begin{proof}\leanok By definition: the \texttt{if} branch is false when $s \geq t$.\end{proof}

\begin{theorem}[Fractional kernel is nonneg]\label{thm:fractionalKernel_nonneg}
\lean{CombinedEnergySpace.fractionalKernel_nonneg}
\leanok
\uses{def:fractionalKernel}
$K_H(t,s) \geq 0$ for $H \geq 1/2$ and $s < t$.
\end{theorem}
\begin{proof}\leanok From \texttt{Real.rpow\_nonneg} applied to $t - s > 0$.\end{proof}

\section{The Combined Energy Space}

\begin{defn}[Combined energy space]\label{def:CombinedEnergySpace}
\lean{CombinedEnergySpace}
\leanok
The product space $\mathcal{H}_{VL} = H_W \times \tilde{H}_J$ where $H_W = L^2([0,T] \times \Omega, \mathcal{P}, \sigma^2 ds \otimes d\mathbb{P})$ is Hilbert (continuous integrands, with divergence $\delta_W(u) = \sigma \int u_t\,dW_t$) and $\tilde{H}_J = L^p([0,T] \times \mathbb{R} \times \Omega, dt \otimes \nu(dz) \otimes d\mathbb{P})$ is Banach (jump integrands, with divergence $\tilde{\delta}_J(h) = \int\!\!\int h(t,z)\,\tilde{N}(dt,dz)$). Carries centering conditions, expectation linearity, H\"older and duality pairings~[1, Definition~4.1].
\end{defn}

\section{Scaling Law for Fractional Kernels}

\begin{defn}[Scaling exponent]\label{def:scalingExponent}

\end{theorem}

\section{Hedging Decomposition}

\begin{defn}[Hedging decomposition]\label{def:HedgingDecomposition}
\lean{CombinedEnergySpace.HedgingDecomposition}
\leanok
\uses{def:CombinedEnergySpace}
The three-level hedge~[1, Proposition~7.2]: discounted claim $= \mathbb{E}^\mathbb{Q}[\text{claim}] + \delta_W(\text{brownian}) + \tilde{\delta}_J(\nu\text{-singular}) + \tilde{\delta}_J(\text{defect})$. Level~1: stock + variance swap absorb all continuous risk. Level~2: a L\'evy driver proxy absorbs the $\nu$-singular (first-order) jump exposure. Level~3: options on jump variation absorb the $\nu$-regular defect. The residual after all three levels is zero---the boundary of hedgeability is the defect $\Gamma_V^f$ itself.
\end{defn}

\begin{theorem}[Hedging residual is jump-only]\label{thm:continuous_hedge_absorbs_brownian}

The continuous hedge absorbs ALL Brownian risk; the remaining risk is entirely in the Poisson integrals~[1, Proposition~7.2(a)]. \end{theorem}
